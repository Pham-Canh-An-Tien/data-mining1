def fill_na(df):
    freq_port = df.Embarked.dropna().mode()[0]
    df['Embarked'] = df['Embarked'].fillna(freq_port)
    df['Fare'].fillna(df['Fare'].dropna().median(), inplace=True)
    df['Age'].fillna(df['Age'].dropna().median(), inplace=True)
